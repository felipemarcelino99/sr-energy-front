# Mock Layer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Install MSW + a mock auth toggle so the app is fully navigable without a backend, reverting to real APIs by setting `VITE_MOCK=false`.

**Architecture:** MSW v2 intercepts all Axios REST calls at the network layer — no service file changes needed. Auth bypasses Supabase via an env-flag branch in `auth.service.ts`. Seed data lives in typed `src/mocks/data/` files; handlers in `src/mocks/handlers/`. Everything activates via `VITE_MOCK=true` in `.env`.

**Tech Stack:** MSW v2 (`msw`), Vite env vars (`import.meta.env.VITE_MOCK`), TypeScript, React 19

## Task 4: Mock Auth Service

**Files:**
- Modify: `src/services/auth.service.ts`

The current file uses `supabase.auth` for everything. We add a mock branch guarded by `VITE_MOCK`. The mock stores the session in `localStorage` under key `mock_auth_user`.

Mock credentials:
- `admin@sr.com` / `123456` → role: `admin`, name: `Admin SR`
- `manager@sr.com` / `123456` → role: `manager`, name: `Gerente SR`
- `employee@sr.com` / `123456` → role: `employee`, name: `Funcionário SR`

- [ ] **Step 1: Replace auth.service.ts**

```ts
import { supabase } from '@/services/supabase'
import type { AuthUser, Role } from '@/models/auth.model'

// ─── Mock auth (VITE_MOCK=true only) ────────────────────────────────────────
// TO REVERT: remove this block and keep only the real auth functions below.

const MOCK_USERS: Record<string, { password: string; user: AuthUser }> = {
  'admin@sr.com': {
    password: '123456',
    user: { id: 'mock-admin-001', email: 'admin@sr.com', name: 'Admin SR', role: 'admin' },
  },
  'manager@sr.com': {
    password: '123456',
    user: { id: 'mock-manager-001', email: 'manager@sr.com', name: 'Gerente SR', role: 'manager' },
  },
  'employee@sr.com': {
    password: '123456',
    user: {
      id: 'mock-employee-001',
      email: 'employee@sr.com',
      name: 'Funcionário SR',
      role: 'employee',
    },
  },
}
const MOCK_SESSION_KEY = 'mock_auth_user'

// ─── Exported functions ──────────────────────────────────────────────────────

export async function signIn(email: string, password: string): Promise<AuthUser> {
  if (import.meta.env.VITE_MOCK === 'true') {
    const entry = MOCK_USERS[email]
    if (!entry || entry.password !== password) throw new Error('Credenciais inválidas')
    localStorage.setItem(MOCK_SESSION_KEY, JSON.stringify(entry.user))
    return entry.user
  }

  const { data, error } = await supabase.auth.signInWithPassword({ email, password })
  if (error) throw new Error(error.message)
  const user = data.user
  const role = (user.user_metadata?.role ?? 'employee') as Role
  const name = (user.user_metadata?.name ?? user.email ?? '') as string
  return { id: user.id, email: user.email!, name, role }
}

export async function signOut(): Promise<void> {
  if (import.meta.env.VITE_MOCK === 'true') {
    localStorage.removeItem(MOCK_SESSION_KEY)
    return
  }

  const { error } = await supabase.auth.signOut()
  if (error) throw new Error(error.message)
}

export async function getSession(): Promise<AuthUser | null> {
  if (import.meta.env.VITE_MOCK === 'true') {
    const raw = localStorage.getItem(MOCK_SESSION_KEY)
    return raw ? (JSON.parse(raw) as AuthUser) : null
  }

  const { data } = await supabase.auth.getSession()
  const session = data.session
  if (!session) return null
  const user = session.user
  const role = (user.user_metadata?.role ?? 'employee') as Role
  const name = (user.user_metadata?.name ?? user.email ?? '') as string
  return { id: user.id, email: user.email!, name, role }
}
```

import { http, HttpResponse } from 'msw'
import { employees as seed } from '@/mocks/data/employees'
import type { Employee, EmployeeFormData } from '@/models/employee.model'

let employees = [...seed]

export const employeeHandlers = [
  http.get(`${import.meta.env.VITE_API_BASE_URL}/employees`, () => {
    return HttpResponse.json(employees)
  }),

  http.get(`${import.meta.env.VITE_API_BASE_URL}/employees/:id`, ({ params }) => {
    const found = employees.find((e) => e.id === params.id)
    if (!found) return HttpResponse.json({ message: 'Not found' }, { status: 404 })
    return HttpResponse.json(found)
  }),

  http.post(`${import.meta.env.VITE_API_BASE_URL}/employees`, async ({ request }) => {
    const body = (await request.json()) as EmployeeFormData
    const newEmployee: Employee = {
      id: `emp-${Date.now()}`,
      ...body,
      salary: Number(body.salary),
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    }
    employees.push(newEmployee)
    return HttpResponse.json(newEmployee, { status: 201 })
  }),

  http.put(`${import.meta.env.VITE_API_BASE_URL}/employees/:id`, async ({ params, request }) => {
    const body = (await request.json()) as EmployeeFormData
    const index = employees.findIndex((e) => e.id === params.id)
    if (index === -1) return HttpResponse.json({ message: 'Not found' }, { status: 404 })
    employees[index] = { ...employees[index], ...body, updatedAt: new Date().toISOString() }
    return HttpResponse.json(employees[index])
  }),

  http.delete(`${import.meta.env.VITE_API_BASE_URL}/employees/:id`, ({ params }) => {
    employees = employees.filter((e) => e.id !== params.id)
    return new HttpResponse(null, { status: 204 })
  }),
]
```

import { contracts as seed } from '@/mocks/data/contracts'
import type { Contract, ContractFormData } from '@/models/contract.model'
import type { ExpiringContract } from '@/models/dashboard.model'

let contracts = [...seed]

function toExpiringContract(c: Contract): ExpiringContract {
  const today = new Date()
  const end = new Date(c.endDate)
  const daysUntilExpiry = Math.ceil((end.getTime() - today.getTime()) / (1000 * 60 * 60 * 24))
  return { id: c.id, clientName: c.clientName, expiresAt: c.endDate, daysUntilExpiry }
}

export const contractHandlers = [
  http.get(`${import.meta.env.VITE_API_BASE_URL}/contracts`, () => {

  }),

  // Must be registered BEFORE /contracts/:id to avoid :id matching "expiring"
  http.get(`${import.meta.env.VITE_API_BASE_URL}/contracts/expiring`, () => {
    const expiring = contracts
      .map(toExpiringContract)
      .filter((c) => c.daysUntilExpiry >= 0 && c.daysUntilExpiry <= 30)
    return HttpResponse.json(expiring)
  }),

  http.get(`${import.meta.env.VITE_API_BASE_URL}/contracts/:id`, ({ params }) => {

    // yield manually. Do NOT reorder these handlers or the report GET will break.
    if (params.id === 'report') return
    const found = jobs.find((j) => j.id === params.id)

    return HttpResponse.json(newJob, { status: 201 })

import { http, HttpResponse } from 'msw'
import { notifications as seed } from '@/mocks/data/notifications'
import type { Notification } from '@/models/notification.model'

let notifications = [...seed]

export const notificationHandlers = [
  http.get(`${import.meta.env.VITE_API_BASE_URL}/notifications`, () => {
    return HttpResponse.json(notifications)
  }),

  http.patch(`${import.meta.env.VITE_API_BASE_URL}/notifications/read-all`, () => {
    notifications = notifications.map((n) => ({ ...n, read: true }))
    return new HttpResponse(null, { status: 204 })
  }),

  http.patch(`${import.meta.env.VITE_API_BASE_URL}/notifications/:id/read`, ({ params }) => {
    const index = notifications.findIndex((n) => n.id === params.id)
    if (index === -1) return HttpResponse.json({ message: 'Not found' }, { status: 404 })
    notifications[index] = { ...notifications[index], read: true }
    return HttpResponse.json(notifications[index])
  }),
]
```

> **Important:** `read-all` handler is registered **before** `/:id/read` to ensure MSW matches the literal path first, preventing `"read-all"` from being treated as an id parameter.
